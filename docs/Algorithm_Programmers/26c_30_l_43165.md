class Solution {
  int target;
  int answer = 0;
  int[] numbers;

  public int solution(int[] numbers, int target) {
    this.numbers = numbers;
    this.target = target;

    dfs(0, 0);

    return answer;

  }

  public void dfs(int index, int result) {
    if (index == numbers.length) {
      if (result == target) answer++;
      return;
    }
    dfs(index+1, result-numbers[index]);
    dfs(index+1, result+numbers[index]);
  }
}
